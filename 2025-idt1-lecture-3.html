Program structure

  • setup()
  • loop()

// Give pin 13 a name.
int led = 13;

// The setup routine runs once.
void setup() {
  // Initialize the digital pin as an output.
  pinMode(led, OUTPUT);
}

// The loop routine runs over and over again forever.
void loop() {
  // turn the LED on
  digitalWrite(led, HIGH);
  // wait for a second
  delay(1000);
  // turn the LED off
  digitalWrite(led, LOW);
  // wait for a second
  delay(1000);
}
